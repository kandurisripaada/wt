    /* Grid Container */
    .grid-container {
      display: grid;                         /* Enables CSS Grid */
      grid-template-columns: 1fr 1fr 1fr; /* Creates 3 equal-width columns */
      grid-template-rows: auto;              /* Rows adjust based on content */
      gap: 10px;                             /* Space between grid items */
      padding: 20px;
      background-color: #f0f0f0;
    }

    /* Grid Items */
    .grid-item {
      background-color: #4CAF50;             /* Green background */
      color: white;
      padding: 20px;
      font-size: 1.2em;
      text-align: center;
      border-radius: 8px;
    }

    /* Demonstrating specific positioning */
    .item1 { grid-column: span 2; }          /* Item spans 2 columns */
    .item2 { grid-row: span 2; }             /* Item spans 2 rows */
